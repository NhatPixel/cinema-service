# --- Các lệnh SQL ---

# SQL tạo bảng
CREATE TABLE notifications (
    id CHAR(36) NOT NULL,
    user_id VARCHAR(50) NOT NULL,
    title VARCHAR(255),
    content TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    expires_at DATETIME NULL,
    PRIMARY KEY (id),
    INDEX idx_user_id (user_id)
);

# Lắng nghe thông báo (SSE)
Method: GET
URL: http://localhost:8080/notifications/stream?user_id=user-123
(giữ kết nối, sẽ nhận event khi có thông báo mới)

# Tạo 1 thông báo (single)
Method: POST
URL: http://localhost:8080/notifications
Body (JSON):
{
  "user_id": "user-123",
  "title": "Khuyến mãi",
  "content": "Giảm 50% vé xem phim cuối tuần",
  "is_read": false
}

# Tạo nhiều thông báo (bulk)
Method: POST
URL: http://localhost:8080/notifications/bulk
Body (JSON):
[
  {
    "user_id": "user-123",
    "title": "Ưu đãi A",
    "content": "Nội dung A",
    "is_read": false
  },
  {
    "user_id": "user-456",
    "title": "Ưu đãi B",

    "is_read": false
  }

Method: PUT
